Added the quantum reservoir functional with CUDA and QuTiP backend support, plus documentation coverage.

Example:
```python
from qilisdk.backends import QutipBackend
from qilisdk.functionals.quantum_reservoirs import ReservoirPass, ReservoirInput, QuantumReservoir
from qilisdk.analog import Schedule, Z, X
from qilisdk.digital import Circuit, U2
from qilisdk.core import ket, QTensor
import numpy as np


pre_processing = Circuit(2)
pre_processing.add(U2(1, phi=ReservoirInput("phi_1", 0.1), gamma=ReservoirInput("gamma_1", 0.1)))

res_pass = ReservoirPass(
    reservoir_dynamics=Schedule(
        hamiltonians={
            "h": Z(0) + Z(1) + Z(0) * Z(1) + 0.5 * (X(0) + X(1))
        }, 
        total_time=1,
        dt=0.1
    ),
    measured_observables=[Z(0), Z(1), Z(0) * Z(1)],
    pre_processing=pre_processing,
    qubits_to_reset=[1]
)



resv = QuantumReservoir(
    initial_state=(np.random.rand() * ket(0, 0) + np.random.rand() * ket(1, 1)).unit(),
    reservoir_pass=res_pass,
    input_per_layer=[
        {"phi_1": 0.2, "gamma_1": 0.1 },
        {"phi_1": 0.3, "gamma_1": 0.2 },
        {"phi_1": 0.4, "gamma_1": 0.3 },
        {"phi_1": 0.5, "gamma_1": 0.4 },
        ],
    store_final_state=True,
    store_intermideate_states=True,
)
backend = QutipBackend()
results = backend.execute(resv)

```
